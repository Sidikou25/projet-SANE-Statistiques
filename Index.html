# 1. Covariance, corrlation et dtermination
cat("\nExercice 2 :\n")
cat("1. Calculs bivaris :\n")
pairs <- combn(quant_vars, 2)
for (i in 1:ncol(pairs)) {
  var1 <- pairs[1, i]
  var2 <- pairs[2, i]
  cat("\nVariables :", var1, "et", var2, "\n")
  cov_val <- cov(data[[var1]], data[[var2]])
  cor_val <- cor(data[[var1]], data[[var2]])
  cat("Covariance :", cov_val, "\n")
  cat("Corrlation :", cor_val, "\n")
  cat("Coefficient de dtermination :", cor_val^2, "\n")
  cat("Interprtation : La relation entre", var1, "et", var2, "est", 
      ifelse(abs(cor_val) > 0.7, "forte", "faible"), "et", 
      ifelse(cor_val > 0, "positive", "ngative"), "\n")
}
